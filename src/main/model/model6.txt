Users

email
password

name (e.g. "Home", "Work", etc.)
street_address
city
state
country
zip_code

product_id (primary key)
name
description
price
quantity

supplier_id (foreign key to Suppliers table)
Categories
category_id (primary key)
name
parent_category_id (foreign key to Categories table)
Suppliers
supplier_id (primary key)
name
email
phone_number

order_id (primary key)
user_id (foreign key to Users table)
order_date
status
total_price
shipping_cost
tax

order_item_id (primary key)

quantity

subtotal_price
Payments
payment_id (primary key)

payment_date
payment_method
card_number
card_holder_name

cart_id (primary key)

date_added

Promotions

end_date
Coupons
coupon_id (primary key)

end_date
Order_Coupons
